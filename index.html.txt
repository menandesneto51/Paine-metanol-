  <script>
    const TZ = 'America/Cuiaba';
    
    function atualizarMomento() {
      const el = document.getElementById('ultimaAtualizacao');
      const agora = new Date();
      
      // 1. Formata apenas a data
      const fmtData = new Intl.DateTimeFormat('pt-BR', {
        dateStyle: 'long',
        timeZone: TZ
      });
      
      const dataFormatada = fmtData.format(agora);
      const horaFixa = "às 10:00";
      
      // 2. Define o texto visível
      el.textContent = `${dataFormatada} ${horaFixa}`;
      
      // 3. Define o atributo 'datetime' para 10:00 no fuso de Cuiabá (UTC-4)
      // Pega as partes da data no fuso correto
      const partesData = new Intl.DateTimeFormat('en-US', {
        year: 'numeric',
        month: 'numeric',
        day: 'numeric',
        timeZone: TZ
      }).formatToParts(agora).reduce((acc, part) => {
        acc[part.type] = part.value;
        return acc;
      }, {});

      // Formata como YYYY-MM-DD
      const ano = partesData.year;
      const mes = partesData.month.padStart(2, '0');
      const dia = partesData.day.padStart(2, '0');
      
      // Cria a string ISO para 10:00 no fuso -04:00 (Cuiabá)
      const isoString = `${ano}-${mes}-${dia}T10:00:00-04:00`;
      el.setAttribute('datetime', isoString);
    }
    
    atualizarMomento();
    
    document.getElementById('btnAtualizarHora')
      .addEventListener('click', atualizarMomento);
  </script>
